    实现常见的字符串 API

    # 和查找相关的方法

    # indexOf

    接受两个参数，第一个参数表示查找目标（注意这里可能不止是单个字符），第二个参数表示开始查找的位置。

    思路：这里用双指针解决。首先从 startIndex 开始，依次遍历原字符串的每一个字符，定位到第一个和目标字符串首字符相等的字符，这里作为第一个指针，接着第二个指针依次遍历该字符往后的字符，按位比较是否和目标字符串的后续字符相等，如果都相等，说明已经找到目标；如果中间有一个不相等，则不是目标，接着遍历原字符串下一个等于目标字符串首字符的字符。

    String.prototype.indexOf = function (target,startIndex = 0){
    let str = this
    for(let i = startIndex;i < str.length;i ++){
        let flag = true
        if(str[i] == target[0]){            
            for(let j = i + 1;j < target.length + i;j ++){
                if(str[j] != target[j-i]){ 
                    flag = false
                    break
                }
            }
            if(flag){
                return i               
            } 
        }
    }
    return -1
}

    # includes

    思路：思路和 indexOf 一样，只不过不是返回索引，而是返回布尔值。

    String.prototype.includes = function(target){
    let str = this
    let flag = true
    for(let i = 0;i < str.length;i ++){
        if(str[i] === target[0]){
            for(let j = i + 1;j < target.length + i;j ++){
                if(str[j] !== target[j-i]){
                    flag = false
                    break
                }
            }
            if(flag){
                return true
            }
        }
    }
    return false
}

    # startsWith、endsWith

    String.prototype.startsWith = function(target){
    let str = this
    let flag = true
    for(let i = 0;i < target.length;i ++){
        if(str[i] !== target[i]){
            flag = false
            break
        }
    }
    return flag
}

String.prototype.endsWith = function(target){
    let str = this
    let flag = true
    let len = str.length - target.length
    for(let i = len;i < str.length;i ++){
        if(str[i] !== target[i - len]){
            flag = false
            break
        }
    }
    return flag
}

    # 和修改相关的方法

    # split

    思路：重点是遍历字符串的时候，如何判断当前遍历到的字符确实属于分隔符的首字符

    // 'da1wd123bkk123vnb123mvlp'  [da1wd,bkk,vnb,mvlp]

String.prototype.split = function(separator){
    let str = this.endsWith(separator) ? this : this + separator
    let res = []
    let each = ''
    let i
    for(i = 0;i < str.length;){
        if(str[i] === separator[0]){
            let flag = true
            for(let j = i + 1;j < i + separator.length;j ++){
             	if(str[j] != separator[j-i]){
                    flag = false
                    break
                }   
            }
            if(flag){
			   res.push(each)
			   each = ''   
			   i =  i + separator.length               
            } else {
                each = each + str[i]
                i = i + 1
            }
        } else {
            each = each + str[i]
            i = i + 1
        }
    }
    return res
}

    # trim

    思路一：正则替换。如果用正则会非常简单，先用 ^ 和 $ 匹配到开头和结尾的字符，\s+ 表示任意多个空格，中间再用 | 逻辑或连接起来即可，最后将匹配到的部分替换为 ""

    String.prototype.trim = function(){
    let str = this
    return str.replace(/^\s+|\s+$/g,"")
}

    思路二：遍历查找

    //  '  abc  d e '  => 'abc  d e'
String.prototype.trim = function(){
    let [startIndex,endIndex] = [-1,-1]
    let arr = this.split('')    
    for(let i = 0;i < arr.length;i ++){
        if(arr[i] != ' '){
            startIndex = i
            break
        }
    }
    if(startIndex >= 0){
        for(let i = arr.length;i >= 0;i --){
            if(arr[i] != ' '){
                endIndex = i
                break
            }
        }
        return arr.slice(startIndex,endIndex + 1).join('')
        
    } else {
        return ''
    }
}

    # concat

    String.prototype.concat = function(...args){
    let str = this
    return args.reduce((acc,cur) => acc + cur,str)
}

    # slice、substring、substr

    slice 基本和 substring 一样，只是参数是负数的时候有区别。这里就不区分了。

    String.prototype.slice = function(start,end){
    let str = this
    let res = ''
    for(let i = start;i < end;i ++){
        res = res + str[i]
    }
    return res
}

    substr 的第二个参数不是末尾元素的索引，而是截取的字符串的长度

    String.prototype.substr = function(start,length){
    let str = this
    let res = ''
    for(let i = start;i < start + length;i ++){
        res = res + str[i]
    }
    return res
}
